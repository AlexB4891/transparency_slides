firm_table <- read_rds("../../../01_DATA/APS/core_sample_assigned.rds")

firm_table <- firm_table %>% 
  mutate(across(c(porcentage_pff,porcentage_ext,porcentage_nac),
                ~.x/100))

firm_table <- firm_table %>% 
  mutate(across(c(total_sales , total_assets,net_profits),
                list(log = ~log(.x + 1)),
                .names = "{.fn}_{.col}")) 

firm_table <- firm_table %>% 
  mutate(
    group_assign = case_when(
      group_assign == "T-Maj" ~ "Majority tax haven",
      group_assign == "T-Min" ~ "Minority tax haven",
      group_assign == "C-Maj" ~ "Majority foreign",
      group_assign == "C-Min" ~ "Minority foreign",
      group_assign == "C-Domestic"  ~ "Majority domestic",
      group_assign == "Domestic" ~ "Domestic"
    )
  )

Task 1: Summary statistics for the core sample

Code 
summ_df <- firm_table %>% 
  filter(anio_fiscal == 2014) %>% 
  summarise(
    across(.cols = c(total_sales,
                     total_income,
                     total_cost_expenses,
                     net_profits,
                     taxable_profits,
                     tax_liability,
                     total_assets),
           .fns = list(
             Mean = ~ mean(.x, na.rm = T),
             SD = ~ sd(.x, na.rm = T),
             p10 = ~ quantile(.x, probs = c(0.1), na.rm = T),
             Median = ~ median(.x, na.rm = T),
             p90 = ~ quantile(.x, probs = c(0.9), na.rm = T)
           ),
           .names = "{.fn}__{.col}"),
    `Mean__Unique firms` = n()) 

reshape_df <- summ_df %>% 
  pivot_longer(cols = everything()) %>% 
  separate(name,into = c("stat","var"),sep = "__") %>% 
  pivot_wider(names_from = stat,
              values_from = value) 


reshape_df <- reshape_df %>% 
  mutate(var = case_when(
    var == "total_sales" ~ "Operational rev.",
    var == "total_income" ~ "Total rev.",
    var == "total_cost_expenses" ~ "Total expenses",
    var == "net_profits" ~ "Gross profit",
    var == "taxable_profits" ~ "CIT tax base",
    var == "tax_liability" ~ "CIT liability",
    var == "total_assets" ~ "Total assets",
    TRUE ~ var
  ))

Task 2: Bubble plot

bubble_df <- firm_table %>% 
  filter(anio_fiscal %in% 2014:2015) %>% 
  select(thp = porcentage_pff,
         rate = tasa_ir,
         anio_fiscal,
         identificacion_informante_anon) %>% 
  split(.$anio_fiscal) %>% 
  imap(rename_with,.cols = c(thp,rate),.fn = ~str_c(.x,.y,sep = "_")) %>% 
  map(select,-anio_fiscal) %>% 
  reduce(full_join) %>% 
  mutate(diff_1415 = rate_2015 - rate_2014) %>% 
  select(diff_1415,thp_2015)

bubble_df <- bubble_df %>% 
  mutate(across(everything(),round,3)) %>% 
  filter(dplyr::between(diff_1415,-0.0005,0.0305)) %>% 
  count(across(everything())) 

n_firms <- bubble_df %>% 
  pull(n) %>% 
  sum(na.rm = T)

  scatter <- bubble_df %>% 
    ggplot() +
    geom_point(aes(x = thp_2015,
                    y = diff_1415,
                    size = n,
                   color = "purple"),
                alpha = 0.5) +
    # geom_text(mapping = aes(y = 0.015,
    #                         x = 0.6,
    #                         label = n_firms)) +
    scale_color_viridis_d(direction = 1) +
    scale_size_continuous(range = c(3,20),
                          guide = "none") +
    scale_shape_discrete(guide = "none") +
    scale_x_continuous(labels = percent) +
    labs(
      # caption =  capti,
         x = "Effective participation in tax havens (2015)",
         y = "Difference in Income Tax Rate 2015-2014",
         color = "") +
    theme_minimal() +
    theme(legend.position = "none",
          plot.caption = element_text(hjust = 0.5),
          axis.title = element_text(size = 8),
          axis.text = element_text(size = 6))
  
  
  scatter

Task 3: Summ stats by group

Code 
summ_df_prel <- firm_table %>% 
  filter(anio_fiscal == 2014,
         str_detect(group_assign,"foreign|tax haven")) %>% 
  group_by(group_assign)


summ_df <- summ_df_prel %>% 
  summarise(
    across(.cols = c(total_sales,
                     total_income,
                     total_cost_expenses,
                     net_profits,
                     taxable_profits,
                     tax_liability,
                     total_assets),
           .fns = list(
             mean = ~ mean(.x, na.rm = T),
             sd = ~ sd(.x, na.rm = T)
           ),
           .names = "{.fn}__{.col}")) 

summ_df_n <- summ_df_prel %>% 
  count() %>% 
  pivot_wider(names_from = group_assign,values_from = n) %>% 
  mutate(across(everything(),number))

reshape_df <- summ_df %>% 
  pivot_longer(cols = -group_assign) %>% 
  separate(name,into = c("stat","var"),sep = "__") %>% 
  mutate(value = round(value,2)) %>% 
  pivot_wider(names_from = c(group_assign,stat),
              values_from = value) 


expresion <- c( "Majority foreign",
                "Majority tax haven",
                "Minority foreign",
                "Minority tax haven") %>% 
  set_names() %>% 
  map(~str_c("str_c(`",.x,"_mean`,'\n(',`",.x,"_sd`,')')")) %>% 
  map(rlang::parse_expr)

reshape_df <- reshape_df %>% 
  transmute(var,!!!expresion)

reshape_df <- reshape_df %>% 
   bind_rows(summ_df_n) %>% 
  mutate(var = case_when(
    var == "total_sales" ~ "Operational rev.",
    var == "total_income" ~ "Total rev.",
    var == "total_cost_expenses" ~ "Total expenses",
    var == "net_profits" ~ "Gross profit",
    var == "taxable_profits" ~ "CIT tax base",
    var == "tax_liability" ~ "CIT liability",
    var == "total_assets" ~ "Total assets",
    TRUE ~ "Unique firms"
  ))

Task 10: Ranking of tax havens

Data construction:

  1. For each firm we take the level of the relationship between the firm and its final shareholders
  2. Direct relationship means relationship at level 1, otherwise it is indirect participation
  3. The ranking is constructed using the values of 2014 for “number of firms that have more than 50% of participation in each country”
Code 
core_sample <- firm_table %>% 
  filter(anio_fiscal == 2014) %>% 
  select(identificacion_informante_anon,group_assign)

participacion <- read_rds("01_DATA/APS/aps_unicos_lista.rds")

d_i_f <- function(firm_data){
  
  if(is.null(firm_data) | nrow(firm_data) == 0){
    NULL
  }else{
    firm_data %>% 
      mutate(direct = as.numeric(nivel_aps == 1),
             codigo_pais_accionista = as.character(codigo_pais_accionista)) %>% 
      group_by(paraiso_fiscal_accionista,
               pais_accionista_unico,
               codigo_pais_accionista,
               direct) %>% 
      summarise(across(porcentaje_efectivo,sum,na.rm = T)) %>% 
      ungroup() %>% 
      mutate(p_e_sqrd = porcentaje_efectivo^2)
  }
  
}
